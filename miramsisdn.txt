{

 if ($resultado eq "G")
   {
    @datos=`mysql $database -e "$query"`;
    foreach $line(@datos)
      {
       print $line;
      }
   }


 if ($resultado eq "L")
   {

 if ($database eq "mvno") {$st_consulta=$db_mvno->prepare($query);}
 if ($database eq "mlog") {$st_consulta=$db_mlog->prepare($query);}

 if ($database eq "mivr") {$st_consulta=$db_mivr->prepare($query);}

 print $query."\n";



 $st_consulta->execute();

 $cabecera=$st_consulta->{'NAME'};
 $campos = $st_consulta->{'NUM_OF_FIELDS'};

 print "***".$descripcion."***\n";

 if ($resultado eq "L")
  {
    while (@datos=$st_consulta->fetchrow_array() )
      {
       for ($n=0; $n<$campos; $n++)
         {
          $$cabecera[$n]=$$cabecera[$n]."                                        ";
          $$cabecera[$n]=substr($$cabecera[$n],0,35);

          print $$cabecera[$n]."|".@datos[$n]."\n";
         }
       print "------------------------------------------------------------------------------------\n";
       }
  }

if ($resultado eq "G")
  {
       for ($n=0; $n<$campos; $n++)
         {
          print $$cabecera[$n]."\t";
         }
       print "\n";

    while (@datos=$st_consulta->fetchrow_array() )
      {
       for ($n=0; $n<$campos; $n++)
         {
          print @datos[$n]."\t";
         }
         print "\n";
       }

  }

$st_consulta->finish();

 print "\n";

}
